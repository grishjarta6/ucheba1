#include <iostream>
using namespace std;

const int INF = 1e9;

void bfs_queue() {
    int n, m, s;
    if (!(cin >> n >> m >> s)) return;

    vector<vector<int> > graph(n + 1);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        graph[u].push_back(v);
        graph[v].push_back(u);
    }

    // массив расстояний, заполненный -1
    vector<int> dist(n + 1, -1);
    queue<int> q;

    // инит стартовой вершины
    dist[s] = 0;
    q.push(s);

    // bfs
    while (!q.empty()) {
        int u = q.front();
        q.pop();

        for (int v : graph[u]) {
            if (dist[v] == -1) { // если вершина еще не посещена
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }

    // вывод результатов
    for (int i = 1; i <= n; ++i) {
        cout << dist[i] << " ";
    }
    cout << "\n";
}

void bunkers() {
    int n, k;
    if (!(cin >> n >> k)) return;

    vector<int> dist(n + 1, -1);
    queue<int> q;

    // высчитываем номера бункеров с выходами
    for (int i = 0; i < k; ++i) {
        int exit_node;
        cin >> exit_node;
        dist[exit_node] = 0; // расстояние для бункера с выходом будет = 0
        q.push(exit_node);   // все источники добавляются в очередь
    }

    int m;
    cin >> m;

    vector<vector<int> > bunkers(n + 1);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        bunkers[u].push_back(v);
        bunkers[v].push_back(u);
    }

    // bfs будет проходиться от выхода во все вершины до конца туннелей бункеров
    while (!q.empty()) {
        int u = q.front();
        q.pop();

        for (int v : bunkers[u]) {
            if (dist[v] == -1) { // если в бункер еще не был посещен
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }

    for (int i = 1; i <= n; ++i) {
        cout << dist[i] << (i == n ? "" : " ");
    }
    cout << "\n";
}

void zero_one_bfs() {
    int n, m, s;
    if (!(cin >> n >> m >> s)) return;

    // хранит числа попарно: сосед1, вес1, сосед2, вес2, ...
    vector<vector<int> > graph(n + 1);
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        graph[u].push_back(v); // сосед
        graph[u].push_back(w); // вес ребра

        graph[v].push_back(u);
        graph[v].push_back(w);
    }

    vector<int> dist(n + 1, -1);
    deque<int> q;

    dist[s] = 0;
    q.push_back(s);

    // bfs для 0-1 вставляет нулевые дороги в начало, которые быстро обработать нужно, а единичные в конец
    while (!q.empty()) {
        int u = q.front();
        q.pop_front();

        // шаг = 2 по массиву, т. к. элементы лежат парами сосед и вес
        for (int i = 0; i < graph[u].size(); i += 2) {
            int v = graph[u][i];
            int w = graph[u][i + 1];

            // если вершина не посещена или нашли путь короче
            if (dist[v] == -1 || dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;

                if (w == 0) {
                    q.push_front(v);
                } else {
                    q.push_back(v);
                }
            }
        }
    }

    for (int i = 1; i <= n; ++i) {
        cout << dist[i] << (i == n ? "" : " ");
    }
    cout << "\n";

}
